country_flavor = {
	type = country

	text = {
		localization_key = rank_kingdom_protectorate_british_republic
		trigger = {
			
			OR = {
				country_rank_is_empire = yes
				country_rank_is_kingdom = yes
			}
			government_type = government_type:republic
			has_reform = government_reform:the_protectorate
			OR = {
				has_or_had_tag = ENG
				has_or_had_tag = GBR
				culture.language = language:english_language
			}
		}
	}
	
	text = {
		localization_key = rank_kingdom_commonwealth_british_republic
		trigger = {

			OR = {
				country_rank_is_empire = yes
				country_rank_is_kingdom = yes
			}
			government_type = government_type:republic
			NOT = { has_reform = government_reform:the_protectorate }
			OR = {
				has_or_had_tag = ENG
				has_or_had_tag = GBR
				culture.language = language:english_language
			}
		}
	}
}
