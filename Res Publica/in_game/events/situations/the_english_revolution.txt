the_english_revolution.100 = {
	type = country_event
	category = situation_event	

	title = the_english_revolution.100.title
	desc = the_english_revolution.100.desc
	historical_info = the_english_revolution_historical_info
	
	image = "gfx/interface/illustrations/situation/the_english_revolution.dds"

	
	option = {
		name = the_english_revolution.100.a
		
		
		remove_country_modifier = the_republic_is_restless
		if = {
			limit = {
				has_country_modifier = noble_lands_impounded
			}
			remove_country_modifier = noble_lands_impounded
		}
		if = {
			limit = {
				has_country_modifier = instrument_enacted
			}
			remove_country_modifier = instrument_enacted
		}
	}
}
